00014     Node lol = NLS::WZ::Top["somenode"]["SomeOtherNode"]["What fun!"];

00022     string lol = somenode;
00023     int foo = somenode;
00024     Sprite wat = somenode;

00026     for (auto it = somenode.Begin(); it != somenode.End(); it++) {
00027         it->first;//Name of the node

00029     }
00030     Don't use the assignment operators.
00031     Those are intended for the wz loading library only.
00032     If you use them the wz data will no longer be what it used to be.
00033     data is the only variable member of a Node.
